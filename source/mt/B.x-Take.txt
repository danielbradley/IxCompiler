~!c/ixcompiler.Take.c~
#include "ixcompiler.h"

static void* stash[100];
~

~c/ixcompiler.Take.c~
void** Give( void* pointer )
{
    void** tmp = stash;

    while ( *tmp ) tmp++;

    *tmp = pointer;

    return tmp;
}
~

~c/ixcompiler.Take.c~
void* Take( ANY given )
{
    void** _given = (void**) given;

    void* keeper = *_given; *_given = null;

    return keeper;
}
~

~c/ixcompiler.Take.c~
void Swap( ANY _one, ANY _two )
{
    void** one = _one;
    void** two = _two;
    void*  tmp;

    tmp  = *one;
    *one = *two;
    *two =  tmp;
}
~
